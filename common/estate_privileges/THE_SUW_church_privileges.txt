#THE_SUW扩张热枕
THE_SUW_estate_church_sennsounetsujyo = {
	icon = privilege_recruit_inouisitor
	max_absolutism = 0
	conditional_modifier = { trigger = { has_absolutism_reduction_for_estate_privileges = yes } modifier = { max_absolutism = 1 } }
	is_valid = {
		OR = {
			has_dlc = "Emperor"
			has_dlc = "Lions of the North"
		}
		NOT = { religion = confucianism }
		OR = {
			has_country_flag = improved_expansionist_zealotry
			any_known_country = {
				NOT = { religion = root }
			}
			has_country_modifier = recently_force_converted_protagonist_timer
			has_country_modifier = recently_annexed_other_religion_timer
		}
		has_country_flag = THE_SUW_1_3_inquisitor_flag
	}
	can_select = {
		OR = {
			has_country_flag = improved_expansionist_zealotry
			any_known_country = {
				NOT = { religion = root }
			}
		}
		NOT = { has_estate_privilege = estate_church_inwards_perfection }
		has_country_flag = THE_SUW_1_3_inquisitor_flag
	}
	on_granted = {
		if = {
			limit = { has_government_attribute = free_expansionist_zealotry }
			custom_tooltip = estate_church_for_the_faith_effect_tooltip_a
		}
		else = {
			custom_tooltip = estate_church_for_the_faith_effect_tooltip
		}
		custom_tooltip = THE_SUW_estate_church_sennsounetsujyo_tt
	}
	on_revoked = {
		if = {
			limit = { has_government_attribute = free_expansionist_zealotry }
			custom_tooltip = estate_church_for_the_faith_effect_tooltip_a
		}
		else = {
			custom_tooltip = estate_church_for_the_faith_effect_tooltip
		}
		custom_tooltip = THE_SUW_estate_church_sennsounetsujyo_tt
	}
	benefits = {
		war_exhaustion = -0.05
	}
	conditional_modifier = {
		trigger = {
			any_known_country = {
				NOT = { religion = root }
				war_with = root
			}
			NOT = { has_country_flag = improved_expansionist_zealotry }
		}
		modifier = {
			church_loyalty_modifier = 0.1
			land_morale = 0.05
		}
	}
	conditional_modifier = {
		trigger = {
			any_known_country = {
				NOT = { religion = root }
				war_with = root
			}
			has_country_flag = improved_expansionist_zealotry
		}
		modifier = {
			church_loyalty_modifier = 0.1
			land_morale = 0.15
			war_exhaustion = -0.05
			max_absolutism = 5
		}
	}
	conditional_modifier = {
		trigger = {
			has_government_attribute = free_expansionist_zealotry
		}
		modifier = {
			max_absolutism = 5
		}
	}
	penalties = {
	}
	conditional_modifier = {
		trigger = { has_government_mechanic = russian_modernization_mechanic }
		modifier = { monthly_russian_modernization = -0.03 }
	}
	conditional_modifier = {
		trigger = { 
			any_known_country = {
				NOT = { religion = root }
				war_with = root
			}
			has_country_flag = THE_SUW_1_3_inquisitor_flag 
			current_age = age_of_reformation
		}
		modifier = {  
			land_morale = 0.05
		}
	}
	ai_will_do = {
		factor = 1
	}
}
