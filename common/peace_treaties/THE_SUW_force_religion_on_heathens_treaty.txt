#Force Religion on Heathens

THE_SUW_force_religion = {
	category = 6
	power_projection = vassalized_rival
	power_cost_base = 0
	prestige_base = 1
	ae_base = 0
	warscore_cost = {
		all_provinces = 1.0
		no_provinces = 0.0
		owner_keeps = 0.0
		conquered = 0.0
		returned_core = 0.0
		released_nation = 0.0
		cancelled_subject = 0.0
		concede_colonial_region = 0.0
	}
	warscore_cap = -1
	requires_demand_independence = yes
	is_make_subject = no
	requires_is_allowed = no
	
	is_visible = {
		OR = {
			AND = {
				has_country_flag = THE_SUW_kan_1_7_flag
				NOT = { religion_group = FROM }
			}
		}
	}
	is_allowed = {
		FROM = {
			NOT = { tag = PAP }	#Pope is always immune to that
			NOT = { religion_group = ROOT }
		}
	}
	effect = {
		FROM = { 
			change_religion = ROOT
			force_converted = yes
			capital_scope = {
				change_religion = ROOT
			}
		}
	}
	ai_weight = { # Defaults to always evaluate to 0
		export_to_variable = {
			variable_name = ai_value # Mandatory to have entry called "ai_value". Its value at the end of the ai_weight scope is what will be used
			value = 10
		}
		limit = {
			always = no	#The AI should be not use this treaty or else they will never expand
		}
	}
}
