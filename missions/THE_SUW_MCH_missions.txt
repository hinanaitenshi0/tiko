THE_SUW_MCH_mission_branch_5 = {
	slot = 5
	generic = no
	ai = yes
	has_country_shield = yes
	potential = { tag = SU0 } 

	THE_SUW_MCH_mission_5_2 = {
		icon = mission_unknown_mission
		position = 2
		required_missions = { THE_SUW_MCH_mission_3_1 } 
		provinces_to_highlight = {
			region = manchuria_region
			owned_by = ROOT
			NOT = { has_tax_building_trigger = yes }
		} 

		trigger = {
			if = {
				limit = { estate_loyalty = { estate = estate_church loyalty = 50 } }
				num_of_owned_provinces_with = { region = manchuria_region has_tax_building_trigger = yes value = 10 }
			}
			else_if = {
				limit = { estate_loyalty = { estate = estate_church loyalty = 40 } }
				num_of_owned_provinces_with = { region = manchuria_region has_tax_building_trigger = yes value = 20 }
			}
			else_if = {
				limit = { estate_loyalty = { estate = estate_church loyalty = 30 } }
				num_of_owned_provinces_with = { region = manchuria_region has_tax_building_trigger = yes value = 30 }
			}
			else = {
				num_of_owned_provinces_with = { region = manchuria_region has_tax_building_trigger = yes value = 40 }
			}
			custom_trigger_tooltip = {
				tooltip = THE_SUW_MCH_mission_5_2_ttt
				always = yes
			}
		} 

		effect = {
			if = {
				limit = { is_subject = yes }
				custom_tooltip = THE_SUW_MCH_mission_5_2_case1_tt
				hidden_effect = {
					every_owned_province = {
						limit = {
							region = manchuria_region
							has_tax_building_trigger = yes
							base_tax = 2
						}
						add_base_tax = -1
						ROOT = {
							overlord = {
								add_treasury = 50
							}
						}
					}
				}
			}
			else = {
				custom_tooltip = THE_SUW_MCH_mission_5_2_case2_tt
				hidden_effect = {
					every_owned_province = {
						limit = {
							region = manchuria_region
							has_tax_building_trigger = yes
						}
						add_base_tax = 1
					}
				}
			}
		} 
	}

	THE_SUW_MCH_mission_5_3 = {
		icon = mission_unknown_mission
		position = 3
		required_missions = { THE_SUW_MCH_mission_5_2 }
